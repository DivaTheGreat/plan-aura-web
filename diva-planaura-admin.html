Setting up Google Sheets + Automatic Welcome Emails

1
Go to sheets.google.com and create a new spreadsheet. Add these column headers in row 1: Name, Email, Source, Date
2
Click Extensions → Apps Script. Delete any existing code and paste the code below.
3
Click Deploy → New Deployment. Type: Web App. Execute as: Me. Who has access: Anyone. Click Deploy and authorise.
4
Copy the Web App URL and paste it into the field below. Every new subscriber will now be added to your Sheet and receive a welcome email from your Gmail automatically.
function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSheet(); sheet.appendRow([data.name, data.email, data.source, new Date().toLocaleDateString()]); MailApp.sendEmail({ to: data.email, subject: "Welcome to Diva | PlanAura ✦", htmlBody: "<div style='font-family:Georgia,serif;max-width:520px;margin:0 auto;color:#1A1A1A'>" + "<h2 style='color:#0B1F3A'>Welcome, " + data.name + "! ✦</h2>" + "<p>Thank you for joining the Diva | PlanAura community. We're so glad you're here.</p>" + "<p>Your free resources are waiting for you. <a href='https://yoursite.com/#freebies' style='color:#D4AF37;font-weight:bold'>Browse all free resources here →</a></p>" + "<p>With grace and order,<br><strong>The PlanAura Team ✦</strong></p>" + "</div>" }); return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
